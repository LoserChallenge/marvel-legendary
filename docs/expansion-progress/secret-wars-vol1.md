async function gainSidekick(destination = "discard") { // "discard" | "hand" | "deckTop"
  if (sidekickDeck.length === 0) { onscreenConsole.log("No Sidekicks remain in the deck."); return null; }
  const sk = sidekickDeck.pop();

}
